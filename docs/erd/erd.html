    users {
        string id PK
        string email
        enum gender "MALE/FEMALE"
        string nickname
        int age
        timestamp created_at
    }
    ootd_photos {
        string id PK
        string user_id FK
        enum photo_type "DAILY/DATE/ME"
        string storage_path
        string image_url
        timestamp uploaded_at
    }
    style_profiles {
        string id PK
        string user_id FK
        string mood_title
        float mood_score
        float warmth_temp
        text ai_comment
        timestamp created_at
    }
    style_keywords {
        string id PK
        string style_profile_id FK
        string keyword
    }
    color_palette {
        string id PK
        string style_profile_id FK
        string hex_code
        int sort_order
    }
    date_moods {
        string id PK
        string style_profile_id FK
        string mood
        bool selected
    }
    matches {
        string id PK
        string user_a_id FK
        string user_b_id FK
        int sync_score
        enum status "PENDING/MATCHED/PASSED"
        timestamp created_at
    }
    chemistry_reports {
        string id PK
        string match_id FK
        int style_score
        int decoration_score
        int mood_score
        int complement_score
        timestamp created_at
    }
    report_points {
        string id PK
        string report_id FK
        enum point_type "MATCH/WARNING"
        text content
    }
    chat_rooms {
        string id PK
        string match_id FK
        timestamp created_at
    }
    messages {
        string id PK
        string chat_room_id FK
        string sender_id FK
        text content
        timestamp sent_at
    }
